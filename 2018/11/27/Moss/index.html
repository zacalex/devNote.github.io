    #!/bin/bash

local_dir="weenix/kernel"
checklist_ourself="checklist/ourself"
checklist="checklist"

if [ "$1" == "1" ]
then 
    echo "copy kernel 1 file to ourself"
    echo "copy kmain to ourself"
    cp $local_dir/main/kmain.c $checklist_ourself/k1/kmain.c
    echo "copy kmutex to ourself"
    cp $local_dir/proc/kmutex.c $checklist_ourself/k1/kmutex.c
    echo "copy kthread to ourself"
    cp $local_dir/proc/kthread.c $checklist_ourself/k1/kthread.c
    echo "copy proc to ourself"
    cp $local_dir/proc/proc.c $checklist_ourself/k1/proc.c
    echo "copy sched to ourself"
    cp $local_dir/proc/sched.c c$checklist_ourselfk1/sched.c
elif [ "$1" == "2" ]
then
    echo "copy kernel 2 file to ourself"
    echo "copy kmain to ourself"
    cp $local_dir/main/kmain.c $checklist_ourself/k2/kmain.c
    echo "copy kmutex to ourself"
    cp $local_dir/proc/kmutex.c $checklist_ourself/k2/kmutex.c
    echo "copy kthread to ourself"
    cp $local_dir/proc/kthread.c $checklist_ourself/k2/kthread.c
    echo "copy proc to ourself"
    cp $local_dir/proc/proc.c $checklist_ourself/k2/proc.c
    echo "copy sched to ourself"
    cp $local_dir/proc/sched.c $checklist_ourself/k2/sched.c

    echo "copy namev to ourself"
    cp $local_dir/fs/namev.c $checklist_ourself/k2/namev.c
    echo "copy open to ourself"
    cp $local_dir/fs/open.c $checklist_ourself/k2/open.c
    echo "copy vfs_syscall to ourself"
    cp $local_dir/fs/vfs_syscall.c $checklist_ourself/k2/vfs_syscall.c
    echo "copy vnode to ourself"
    cp $local_dir/fs/vnode.c $checklist_ourself/k2/vnode.c

elif [ "$1" == "3" ]
then 
    echo "copy kernel 3 file to ourself"
    echo "copy kmain to ourself"
    cp $local_dir/main/kmain.c $checklist_ourself/k3/kmain.c

    echo "copy kmutex to ourself"
    cp $local_dir/proc/kmutex.c $checklist_ourself/k3/kmutex.c
    echo "copy kthread to ourself"
    cp $local_dir/proc/kthread.c $checklist_ourself/k3/kthread.c
    echo "copy proc to ourself"
    cp $local_dir/proc/proc.c $checklist_ourself/k3/proc.c
    echo "copy sched to ourself"
    cp $local_dir/proc/sched.c $checklist_ourself/k3/sched.c
    echo "copy sched to ourself"
    cp $local_dir/proc/sched.c $checklist_ourself/k3/sched.c
    echo "copy sched to ourself"
    cp $local_dir/proc/sched_helper.c $checklist_ourself/k3/sched_helper.c

    echo "copy namev to ourself"
    cp $local_dir/fs/namev.c $checklist_ourself/k3/namev.c
    echo "copy open to ourself"
    cp $local_dir/fs/open.c $checklist_ourself/k3/open.c
    echo "copy vfs_syscall to ourself"
    cp $local_dir/fs/vfs_syscall.c $checklist_ourself/k3/vfs_syscall.c
    echo "copy vnode to ourself"
    cp $local_dir/fs/vnode.c $checklist_ourself/k3/vnode.c

    echo "copy fork to ourself"
    cp $local_dir/proc/fork.c $checklist_ourself/k3/fork.c

    echo "copy access.c to ourself"
    cp $local_dir/api/access.c $checklist_ourself/k3/access.c
    echo "copy syscall to ourself"
    cp $local_dir/api/syscall.c $checklist_ourself/k3/syscall.c

    echo "copy pframe.c to ourself"
    cp $local_dir/mm/pframe.c $checklist_ourself/k3/pframe.c

    echo "copy namev to ourself"
    cp $local_dir/vm/anon.c $checklist_ourself/k3/anon.c
    echo "copy brk to ourself"
    cp $local_dir/vm/brk.c $checklist_ourself/k3/brk.c
    echo "copy mmap to ourself"
    cp $local_dir/vm/mmap.c $checklist_ourself/k3/mmap.c
    echo "copy pagefault to ourself"
    cp $local_dir/vm/pagefault.c $checklist_ourself/k3/pagefault.c
    echo "copy shadow to ourself"
    cp $local_dir/vm/shadow.c $checklist_ourself/k3/shadow.c
    echo "copy vmmap to ourself"
    cp $local_dir/vm/vmmap.c $checklist_ourself/k3/vmmap.c
fi


if [ "$3" == "b" ]
then 
    echo "do moss for kernel "$1" with ref "$2" with codebase"
    ./moss -b checklist/base/codebase.c -d $checklist_ourself/k"$1"/*.c $checklist/"$2"/k"$1"/*.c
else 
    echo "do moss for kernel "$1" with ref "$2" without codebase"
    ./moss -d $checklist_ourself/k"$2"/*.c $checklist/"$2"/k"$2"/*.c
fi
